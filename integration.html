使用示例 

// 1. 初始化整合
const integration = new AIToolHubIntegration({
    agentEndpoint: 'http://localhost:3000/api'
});

// 2. 启动
await integration.init();

// 3. 用户点击"执行"按钮
// → 自动调用 OpenClaw Agent API
// → 通过 Intent Engine 识别意图
// → 执行对应工具
// → 返回结果到前端

// 4. API 调用示例
const result = await fetch('http://localhost:3000/api/execute', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
        tool: 'agent',
        inputs: {
            prompt: '帮我写一段关于AI的文案',
            _meta: { source: 'ai-tool-hub', toolName: 'ChatGPT' }
        }
    })
});
